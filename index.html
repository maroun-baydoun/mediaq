mediaq

Listen to media query updates in JavaScript

Github NPM

Install

npm i mediaq
Or
yarn add mediaq

Use

import { Mediaq } from "mediaq";

const mediaq = Mediaq({
    onUpdate: (e) => console.log(e.name, e.media, e.matches),
    mediaQueries: [{
            name: "mobile",
            media: "only screen and (max-width: 600px)"
        }, 
        {
            name: "desktop",
            media: "only screen and (min-width: 600px)"
        },
    ]
});

mediaq.start();

// When done listening 
mediaq.stop();
